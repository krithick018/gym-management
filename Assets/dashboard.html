Step 2: Serve Static Files Ensure that Express can serve static files like CSS and JS if you plan to move the CSS and JS to separate files in the future. js Copy code // server.js const express = require('express'); const mongoose = require('mongoose'); const bodyParser = require('body-parser'); const path = require('path'); const Registration = require('./models/Registration'); const app = express(); mongoose.connect('mongodb://localhost:27017/registrationDB', { useNewUrlParser: true, useUnifiedTopology: true }); app.use(bodyParser.urlencoded({ extended: true })); app.set('view engine', 'ejs'); app.set('views', path.join(__dirname, 'views')); app.use(express.static(path.join(__dirname, 'public'))); // Serve static files app.post('/joinnow', async (req, res) => { try { const { fullName, Aadhaarcard, email, phoneNumber, weight, options, gender } = req.body; const registration = new Registration({ fullName, Aadhaarcard, email, phoneNumber, weight, options, gender }); await registration.save(); res.redirect('/dashboard'); } catch (error) { res.status(500).send("Error saving registration data: " + error.message); } }); app.get('/dashboard', async (req, res) => { try { const registrations = await Registration.find({}); res.render('dashboard', { registrations }); } catch (error) { res.status(500).send("Error retrieving registration data: " + error.message); } }); app.listen(3000, () => { console.log('Server started on http://localhost:3000'); });
